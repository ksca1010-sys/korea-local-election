## 2. Layer 1: 시간 기준

```javascript
// js/election-calendar.js

// ─── 시간 기준 ───
// 모든 시간 판정은 이 함수를 통해서만.
// 해외 접속(재외국민)도 KST 기준으로 판정.
const getKST = () => {
  const now = new Date();
  const utc = now.getTime() + (now.getTimezoneOffset() * 60000);
  return new Date(utc + (9 * 3600000));
};

// ─── 날짜 상수 ───
// 문자열 비교 금지. 반드시 Date 객체로 정의.
// 모든 시각은 +09:00 (KST) 명시.
const DATES = {
  // 예비후보 등록
  PRE_REG_GOVERNOR:    new Date('2026-02-03T00:00:00+09:00'),  // 시도지사·교육감
  PRE_REG_COUNCIL:     new Date('2026-02-20T00:00:00+09:00'),  // 시도의원·구시의원·장
  PRE_REG_GUN:         new Date('2026-03-22T00:00:00+09:00'),  // 군의원·군수·재보궐

  // 법적 기한
  RESIGNATION_DEADLINE:new Date('2026-03-05T00:00:00+09:00'),  // 공직자 사직
  PROPORTIONAL_RESIGN: new Date('2026-05-04T00:00:00+09:00'),  // 비례대표 사직

  // 선거인명부
  VOTER_LIST_START:    new Date('2026-05-12T00:00:00+09:00'),
  VOTER_LIST_END:      new Date('2026-05-16T23:59:59+09:00'),

  // 후보자 등록
  CANDIDATE_REG_START: new Date('2026-05-14T09:00:00+09:00'),
  CANDIDATE_REG_END:   new Date('2026-05-15T18:00:00+09:00'),  // 이 시각 이후 기호순 전환

  // 선거운동
  CAMPAIGN_START:      new Date('2026-05-21T00:00:00+09:00'),
  CAMPAIGN_END:        new Date('2026-06-02T23:59:59+09:00'),

  // 여론조사 공표금지 (법적 필수)
  PUBLICATION_BAN_START: new Date('2026-05-28T00:00:00+09:00'),

  // 사전투표
  EARLY_VOTE_START:    new Date('2026-05-29T06:00:00+09:00'),
  EARLY_VOTE_END:      new Date('2026-05-30T18:00:00+09:00'),

  // 투표일
  ELECTION_DAY_START:  new Date('2026-06-03T06:00:00+09:00'),
  VOTE_END:            new Date('2026-06-03T18:00:00+09:00'),   // 공표금지 해제 시점

  // 선거 후
  INAUGURATION:        new Date('2026-07-01T00:00:00+09:00'),
};
```

## 3. Layer 2A: 법적 필수 로직 (독립)

```javascript
// ─── 여론조사 공표금지 ───
// 공직선거법 제108조
// 선거일 전 6일(5/28 00:00)부터 투표 마감(6/3 18:00)까지
// 여론조사 결과 공표·보도 금지
// 위반 시 3년 이하 징역 또는 600만원 이하 벌금
//
// ⚠️ 이 함수는:
//   - getCurrentPhase()를 참조하지 않음
//   - 다른 어떤 함수의 반환값에 의존하지 않음
//   - DATES 상수와 getKST()만 사용
//   - UI 숨김이 아니라 데이터 자체를 차단

const isPublicationBanned = () => {
  const now = getKST();
  return now >= DATES.PUBLICATION_BAN_START && now <= DATES.VOTE_END;
};

// 여론조사 데이터 접근 시 반드시 이 함수를 거침
const getFilteredPolls = (allPolls, selection) => {
  // 공표금지 기간이면 데이터 자체를 빈 배열로 반환
  // UI 숨김이 아님 — 개발자 도구에서도 데이터 접근 불가
  if (isPublicationBanned()) {
    return {
      polls: [],
      banned: true,
      notice: '선거법에 따라 여론조사 결과를 표시할 수 없습니다.\n' +
              '공표금지 기간: 5월 28일 00:00 ~ 6월 3일 18:00\n' +
              '근거: 공직선거법 제108조'
    };
  }
  
  // 정상 기간이면 기존 필터링 로직 실행
  const filtered = getPollsForSelection(allPolls, selection);
  return { polls: filtered, banned: false, notice: null };
};

// 뉴스탭 여론조사 하부메뉴도 동일하게 차단
const isNewsSubTabDisabled = (subTabName) => {
  if (subTabName === '여론조사' && isPublicationBanned()) {
    return {
      disabled: true,
      notice: '공표금지 기간에는 여론조사 관련 뉴스를 표시할 수 없습니다.'
    };
  }
  return { disabled: false, notice: null };
};
```

## 4. Layer 2B: 편의 로직 (phase 기반)

```javascript
// ─── 시기 판정 ───
// 법적 의무와 무관한 UX 편의 기능만 담당
// 배너 메시지, 뉴스탭 기본 메뉴, 후보자 정렬 등

const getCurrentPhase = () => {
  const now = getKST();

  if (now < DATES.CANDIDATE_REG_START)  return 'PRE_REGISTRATION';
  if (now <= DATES.CANDIDATE_REG_END)   return 'REGISTRATION';
  if (now < DATES.CAMPAIGN_START)       return 'POST_REGISTRATION';
  if (now < DATES.EARLY_VOTE_START)     return 'CAMPAIGN';
  if (now <= DATES.EARLY_VOTE_END)      return 'EARLY_VOTING';
  if (now < DATES.ELECTION_DAY_START)   return 'PRE_ELECTION_DAY';
  if (now <= DATES.VOTE_END)            return 'ELECTION_DAY';
  if (now < DATES.INAUGURATION)         return 'POST_ELECTION';
  return 'INAUGURATED';
};

// ─── D-day ───
const getDday = () => {
  const now = getKST();
  const election = new Date('2026-06-03T00:00:00+09:00');
  const diff = Math.ceil((election - now) / (1000 * 60 * 60 * 24));
  if (diff > 0) return `D-${diff}`;
  if (diff === 0) return 'D-Day';
  return `D+${Math.abs(diff)}`;
};

// ─── 후보자 정렬 모드 ───
const getCandidateSortMode = () => {
  const now = getKST();
  return now > DATES.CANDIDATE_REG_END ? 'ballot_number' : 'alphabetical';
};

// ─── 뉴스탭 기본 하부메뉴 ───
const getDefaultNewsSubTab = () => {
  const phase = getCurrentPhase();
  if (phase === 'CAMPAIGN' || phase === 'PRE_ELECTION_DAY') return '선거운동';
  return '전체';
};

// ─── 상단 배너 ───
const getBannerConfig = () => {
  const phase = getCurrentPhase();
  const now = getKST();

  // 선거인명부 열람 기간
  if (now >= DATES.VOTER_LIST_START && now <= DATES.VOTER_LIST_END) {
    return {
      show: true,
      text: '선거인명부 열람 기간입니다 (5/12~16)',
      type: 'info',
      link: { label: '선관위 열람', url: 'https://www.nec.go.kr' }
    };
  }

  switch (phase) {
    case 'REGISTRATION':
      return {
        show: true,
        text: '후보자 등록 기간 (5/14~15)',
        type: 'info',
        link: null
      };

    case 'CAMPAIGN':
    case 'PRE_ELECTION_DAY':
      // 공표금지 배너가 우선
      if (isPublicationBanned()) {
        return {
          show: true,
          text: '여론조사 공표금지 기간 (5/28~6/3 18:00)',
          type: 'warning',
          link: null
        };
      }
      return {
        show: true,
        text: '공식 선거운동 기간 (5/21~6/2)',
        type: 'campaign',
        link: null
      };

    case 'EARLY_VOTING':
      return {
        show: true,
        text: '사전투표 진행 중 | 06:00~18:00 | 전국 어디서나',
        type: 'vote',
        link: { label: '투표소 찾기', url: 'https://www.nec.go.kr' }
      };

    case 'ELECTION_DAY':
      if (now <= DATES.VOTE_END) {
        return {
          show: true,
          text: '오늘은 투표일입니다 | 06:00~18:00',
          type: 'vote',
          link: { label: '투표소 찾기', url: 'https://www.nec.go.kr' }
        };
      }
      return {
        show: true,
        text: '투표가 마감되었습니다. 개표가 진행 중입니다.',
        type: 'result',
        link: { label: '개표 현황', url: 'https://www.nec.go.kr' }
      };

    case 'POST_ELECTION':
      return {
        show: true,
        text: '제9회 전국동시지방선거가 종료되었습니다',
        type: 'info',
        link: null
      };

    default:
      return { show: false };
  }
};
```

## 5. Layer 3: 사이트 적용 — 각 컴포넌트별 호출 방법

```javascript
// ─── 여론조사 탭 렌더링 ───
function renderPollTab(region, electionType) {
  // Layer 2A를 먼저 체크 (법적 필수)
  const result = getFilteredPolls(allPollsData, { region, electionType });
  
  if (result.banned) {
    // 데이터가 빈 배열 — 금지 안내만 표시
    showPollBanNotice(result.notice);
    return;
  }
  
  // 정상 기간 — 기존 렌더링
  renderPollCards(result.polls);
  renderPollTrendChart(result.polls);
}

// ─── 뉴스탭 렌더링 ───
function renderNewsTab(region, electionType, subTab) {
  // Layer 2A 체크 — 여론조사 하부메뉴 차단
  const subTabCheck = isNewsSubTabDisabled(subTab);
  if (subTabCheck.disabled) {
    showNewsSubTabNotice(subTabCheck.notice);
    return;
  }
  
  // Layer 2B — 기본 하부메뉴
  if (!subTab) {
    subTab = getDefaultNewsSubTab();
  }
  
  // 기존 렌더링
  renderNewsCards(region, electionType, subTab);
}

// ─── 후보자탭 렌더링 ───
function renderCandidateTab(region, electionType) {
  // Layer 2B — 정렬 모드
  const sortMode = getCandidateSortMode();
  const candidates = getCandidatesForSelection(region, electionType);
  
  if (sortMode === 'ballot_number') {
    candidates.sort((a, b) => a.ballotNumber - b.ballotNumber);
  } else {
    candidates.sort((a, b) => a.name.localeCompare(b.name, 'ko'));
  }
  
  renderCandidateCards(candidates, sortMode);
}

// ─── 개요탭 렌더링 ───
function renderOverviewTab(region, electionType) {
  // 여론조사 미니 위젯도 공표금지 체크
  const pollResult = getFilteredPolls(allPollsData, { region, electionType });
  
  if (!pollResult.banned && pollResult.polls.length > 0) {
    renderPollMiniWidget(pollResult.polls);
  }
  // 공표금지면 미니 위젯 자체를 렌더링하지 않음 (빈 공간도 아님, 아예 없음)
  
  // 나머지 개요 렌더링
  renderOverviewContent(region, electionType);
}

// ─── 상단 배너 렌더링 ───
function renderBanner() {
  const config = getBannerConfig();
  if (!config.show) {
    hideBanner();
    return;
  }
  showBanner(config.text, config.type, config.link);
}

// ─── 페이지 로드 시 ───
function initElectionCalendar() {
  // D-day 업데이트
  document.getElementById('dday').textContent = getDday();
  
  // 배너 표시
  renderBanner();
  
  // 1시간마다 자동 갱신 (자정에 phase 전환 대비)
  setInterval(() => {
    renderBanner();
    document.getElementById('dday').textContent = getDday();
  }, 60 * 60 * 1000);
}
```

## 7. 정당 공천 추적 (캘린더에 넣지 않음)

정당 공천은 고정 날짜가 없으므로 뉴스 자동 수집 + 수동 업데이트로 추적:

```
추적 키워드:
  "민주당 공천" "국민의힘 공천" "조국혁신당 공천" "개혁신당 공천"
  "경선 결과" "단수공천" "전략공천" "공천 확정" "공천 탈락"
  "무소속 출마"

데이터 반영:
  공천 확정   → status: "공천확정"
  경선 진행   → status: "경선참여"  
  공천 탈락 후 무소속 → party: "무소속", status: "출마선언"
  사퇴/포기   → status: "WITHDRAWN"

예상 시기 (참고용, 고정 아님):
  더불어민주당: 적격심사 3월 → 경선 4월 초~중순 → 확정 4월 중순
  국민의힘: 공천심사 3월말~4월 → 확정 4월 중~하순
  조국혁신당: 호남권 중심, 시기 미확정
  개혁신당: 시기 미확정
```
