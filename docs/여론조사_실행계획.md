> 방향: Chart.js 추이 그래프를 빼고, 한줄요약 + 바차트 + 카드로 판세 전달
> 추이 그래프는 나중에 여유 있을 때 추가 (지금은 구현 안 함)
> 바차트도 Chart.js가 아니라 CSS div로 구현 (구현 난이도 최소화)
> docs/ 폴더에 넣고 Claude Code에 "docs/여론조사_실행계획.md를 읽고 구현해줘" 지시

## 2. [1단] 한줄 판세 요약

### 작동 방식

```
가장 최신 조사 1건에서 상위 2명을 뽑고,
격차와 오차범위를 비교해서 한 줄로 판정.

가중평균 안 함. 합산 안 함. 최신 단일 조사 기준.
```

### 판정 로직

```javascript
function generatePollInsight(polls, electionType) {
  // 결과가 있는 가장 최신 조사
  const latest = polls
    .filter(p => p.results?.length > 0)
    .sort((a, b) => new Date(b.surveyDateEnd || b.registeredDate) 
                   - new Date(a.surveyDateEnd || a.registeredDate))[0];
  
  if (!latest) return null;
  
  // 실제 후보만 (모르겠다/무응답 제외)
  const candidates = latest.results
    .filter(r => r.candidateName && 
                 !['모르겠다', '모름/무응답', '무응답', '지지', '없음'].includes(r.candidateName))
    .sort((a, b) => b.support - a.support);
  
  if (candidates.length < 2) return null;
  
  const first = candidates[0];
  const second = candidates[1];
  const gap = first.support - second.support;
  const moe = latest.marginOfError || 3.5; // 오차범위 (없으면 3.5 기본)
  
  let type, badge, badgeColor, text;
  
  if (gap <= moe) {
    type = 'close';
    badge = '접전';
    badgeColor = '#F5A623';
    text = `${first.candidateName} vs ${second.candidateName} — 오차범위 내 격차`;
  } else if (gap <= moe * 2) {
    type = 'competitive';
    badge = '경합';
    badgeColor = '#00BCD4';
    text = `${first.candidateName} ${gap.toFixed(1)}%p 앞서 — 경합 구도`;
  } else {
    type = 'leading';
    badge = '우세';
    badgeColor = '#45B26B';
    text = `${first.candidateName} ${gap.toFixed(1)}%p 우세`;
  }
  
  return {
    type, badge, badgeColor, text,
    source: {
      org: latest.pollOrg,
      date: latest.surveyDateEnd || latest.registeredDate,
      sampleSize: latest.sampleSize,
      marginOfError: moe
    }
  };
}
```

## 3. [2단] 최신 지지율 바 차트 (CSS 순수 구현, Chart.js 안 씀)

### 색상 함수

```javascript
// 정당색 → 바 색상
function getCandidateColor(candidate, electionType) {
  // 1순위: 정당색
  if (candidate.partyCode || candidate.party) {
    const partyColor = getPartyColorByName(candidate.party);
    if (partyColor) return partyColor;
  }
  
  // 2순위: 교육감 성향
  if (electionType === 'superintendent' && candidate.stance) {
    const stanceColors = { '진보': '#1A5CF0', '보수': '#E61E2B', '중도': '#45B26B' };
    return stanceColors[candidate.stance] || '#888880';
  }
  
  // 3순위: 무소속 순번별 고유색
  const independentColors = ['#F5A623', '#7B61FF', '#00BCD4', '#E91E63', '#8BC34A', '#FF7043'];
  return independentColors[candidate._index % independentColors.length] || '#888880';
}

// 정당명 → 색상 (parties.json 기반 + 한글명 매칭)
function getPartyColorByName(partyName) {
  if (!partyName) return null;
  const map = {
    '더불어민주당': '#1A5CF0',
    '국민의힘': '#E61E2B',
    '조국혁신당': '#FF6D2E',
    '개혁신당': '#FF7210',
    '진보당': '#D6001C',
    '녹색정의당': '#00A85A',
    '새로운미래': '#003DA5',
    '기본소득당': '#00C9B7',
  };
  // 부분 매칭 (약칭 대응)
  for (const [name, color] of Object.entries(map)) {
    if (partyName.includes(name) || name.includes(partyName)) return color;
  }
  return null; // 무소속 등
}
```

### 바 차트 렌더링

```javascript
function renderSnapshotBars(poll, electionType) {
  if (!poll?.results?.length) return '';
  
  const candidates = poll.results
    .filter(r => r.candidateName && 
                 !['모르겠다', '모름/무응답', '무응답', '지지', '없음'].includes(r.candidateName))
    .sort((a, b) => b.support - a.support)
    .map((c, i) => ({ ...c, _index: i }));
  
  const maxSupport = candidates[0]?.support || 50;
  
  // 전회 대비 변동 계산용 이전 조사
  const previousPoll = null; // 추후 연결
  
  let html = '<div class="snapshot-bars">';
  
  candidates.forEach((c, index) => {
    const color = getCandidateColor(c, electionType);
    const widthPercent = Math.max((c.support / maxSupport) * 100, 2); // 최소 2%
    
    // 정당/성향 표시
    let partyLabel = c.party || '';
    if (electionType === 'superintendent') {
      const stance = ElectionData.getSuperintendentStance?.(currentRegionKey, c.candidateName);
      if (stance) partyLabel = stance;
    }
    // "무소속"이 연속 반복되면 첫 번째만 표시
    if (partyLabel === '무소속' && index > 0) {
      const prevParty = candidates[index - 1]?.party;
      if (prevParty === '무소속') partyLabel = '';
    }
    
    html += `
      <div class="snapshot-bar-row">
        <div class="snapshot-bar-info">
          <span class="snapshot-name">${c.candidateName}</span>
          <span class="snapshot-party" style="color:${color};">${partyLabel}</span>
          <span class="snapshot-support">${c.support}%</span>
        </div>
        <div class="snapshot-bar-track">
          <div class="snapshot-bar-fill" style="width:${widthPercent}%; background:${color};"></div>
        </div>
      </div>
    `;
  });
  
  html += '</div>';
  
  // 오차범위 해석
  if (candidates.length >= 2 && poll.marginOfError) {
    const gap = candidates[0].support - candidates[1].support;
    const moe = poll.marginOfError;
    
    if (gap <= moe) {
      html += `<div class="moe-interpretation moe-within">
        ⚠️ 1·2위 격차(${gap.toFixed(1)}%p)가 오차범위(±${moe}%p) 안 — 우열을 가릴 수 없습니다
      </div>`;
    } else {
      html += `<div class="moe-interpretation moe-outside">
        격차 ${gap.toFixed(1)}%p — 오차범위(±${moe}%p) 밖의 유의미한 차이
      </div>`;
    }
  }
  
  return html;
}
```

```javascript
function renderPollCards(polls, electionType) {
  const withResults = polls
    .filter(p => p.results?.length > 0)
    .sort((a, b) => new Date(b.surveyDateEnd || b.registeredDate) 
                   - new Date(a.surveyDateEnd || a.registeredDate));
  
  const withoutResults = polls
    .filter(p => !p.results || p.results.length === 0);
  
  let html = '';
  
  // 결과 있는 카드 (각각에 미니 바 차트 포함)
  withResults.forEach(poll => {
    html += renderSinglePollCard(poll, electionType);
  });
  
  // 결과 없는 카드 — 접이식
  if (withoutResults.length > 0) {
    html += `
      <details class="poll-uncollected">
        <summary class="poll-uncollected-summary">
          결과 미수집 조사 (${withoutResults.length}건)
        </summary>
        <div class="poll-uncollected-list">
          ${withoutResults.map(p => renderMinimalPollCard(p)).join('')}
        </div>
      </details>
    `;
  }
  
  return html;
}

function renderSinglePollCard(poll, electionType) {
  const method = poll.method || '';
  const sampleSize = poll.sampleSize ? `n=${poll.sampleSize.toLocaleString()}` : '';
  const moe = poll.marginOfError ? `±${poll.marginOfError}%p` : '';
  const dateStr = formatPollDate(poll);
  
  // 미니 바 차트 (2단 스냅샷보다 작은 버전)
  const miniBars = renderMiniBars(poll.results, electionType);
  
  return `
    <div class="poll-card">
      <div class="poll-card-header">
        <span class="poll-card-org">${poll.pollOrg || '조사기관 미상'}</span>
        ${method ? `<span class="poll-card-tag">${method}</span>` : ''}
        ${sampleSize ? `<span class="poll-card-tag">${sampleSize}</span>` : ''}
      </div>
      <div class="poll-card-meta">
        ${poll.clientOrg ? `의뢰: ${poll.clientOrg}` : ''}
        ${dateStr ? `<br>${dateStr}` : ''}
        ${moe ? `<br>오차범위 ${moe} (95% 신뢰수준)` : ''}
      </div>
      <div class="poll-card-results">
        ${miniBars}
      </div>
      ${poll.nesdcDetailUrl ? `
        <a class="poll-card-link" href="${poll.nesdcDetailUrl}" target="_blank" rel="noopener">
          여심위 원문 보기 →
        </a>
      ` : ''}
    </div>
  `;
}

function renderMinimalPollCard(poll) {

      <div class="poll-card-meta">

      </div>
      ${poll.nesdcDetailUrl ? `
        <a class="poll-card-link" href="${poll.nesdcDetailUrl}" target="_blank" rel="noopener">
          여심위 원문 보기 →
        </a>
      ` : ''}

  `;
}

function renderMiniBars(results, electionType) {
  if (!results?.length) return '';
  
  const candidates = results
    .filter(r => r.candidateName && 
                 !['모르겠다', '모름/무응답', '무응답', '지지', '없음'].includes(r.candidateName))
    .sort((a, b) => b.support - a.support)
    .map((c, i) => ({ ...c, _index: i }));
  
  const max = candidates[0]?.support || 50;
  
  return candidates.map(c => {
    const color = getCandidateColor(c, electionType);
    const w = Math.max((c.support / max) * 100, 2);
    let partyLabel = c.party || '';
    if (electionType === 'superintendent') {
      const stance = ElectionData.getSuperintendentStance?.(currentRegionKey, c.candidateName);
      if (stance) partyLabel = stance;
    }
    
    return `
      <div class="mini-bar-row">
        <span class="mini-bar-name">${c.candidateName}</span>
        <span class="mini-bar-party" style="color:${color};">${partyLabel}</span>
        <span class="mini-bar-value">${c.support}%</span>
        <div class="mini-bar-track">
          <div class="mini-bar-fill" style="width:${w}%; background:${color};"></div>
        </div>
      </div>
    `;
  }).join('');
}
```

```javascript
// app.js의 renderPollTab()을 이것으로 교체

function renderPollTab() {
  const container = document.getElementById('poll-content'); // 또는 실제 컨테이너 ID

      </div>`;
    return;
  }
  
  // 여론조사 데이터 가져오기
  const polls = ElectionData.getLatestPollsForDisplay(
    currentRegionKey, currentElectionType, currentDistrictName

    .filter(p => p.results?.length > 0)
    .sort((a, b) => new Date(b.surveyDateEnd || b.registeredDate)

    html += `<div class="snapshot-source">
      출처: ${latestWithResults.pollOrg} · ${formatDate(latestWithResults.surveyDateEnd || latestWithResults.registeredDate)}
    </div>`;
    html += `</div>`;
  }
  
  // [3단] 개별 조사 카드
  html += `<h4 class="section-label" style="margin-top:20px;">전체 조사 (${polls.length}건)</h4>`;
  html += renderPollCards(polls, currentElectionType);
  
  container.innerHTML = html;
}

function renderNoPollMessage() {
  const messages = {
    governor: '이 지역의 광역단체장 여론조사가 아직 등록되지 않았습니다.',
    superintendent: '이 지역의 교육감 여론조사가 아직 등록되지 않았습니다.',
    mayor: '이 지역의 기초단체장 여론조사가 아직 등록되지 않았습니다.\n해당 시도의 광역단체장 여론조사를 참고하세요.',
    byElection: '이 선거구의 여론조사가 아직 등록되지 않았습니다.',
  };
  
  return `<div class="poll-insight poll-insight-empty">
    ${messages[currentElectionType] || '여론조사가 아직 등록되지 않았습니다.'}
  </div>`;
}

// 날짜 포맷
function formatDate(dateStr) {
  if (!dateStr) return '';
  const d = new Date(dateStr);
  return `${d.getMonth() + 1}/${d.getDate()}`;
}

function formatPollDate(poll) {
  const start = poll.surveyDateStart || '';
  const end = poll.surveyDateEnd || '';
  const pub = poll.publishDate || '';
  
  let text = '';
  if (start && end) {
    text += `${start}~${end} 조사`;
  }
  if (pub) {
    text += ` / ${pub} 공표`;
  }
  return text;
}
```
